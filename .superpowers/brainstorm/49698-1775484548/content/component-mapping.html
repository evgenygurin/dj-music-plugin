Vercel Data Fetching Strategy

Read (Server Components)
// Next.js 16 — ISR revalidation
export const revalidate = 30

// Supabase query in Server Component
const supabase = await createClient()
const { data } = await supabase
  .from('tracks')
  .select('*, track_audio_features_computed(*)')
  .eq('status', 0)
  .range(0, 49)

// Next.js 16 remote cache
async function getStats() {
  'use cache: remote'
  cacheLife({ expire: 60 })
  cacheTag('library-stats')
  ...
}
Write (Server Actions → MCP)
'use server'

// After MCP mutation, revalidate cache
import { revalidateTag } from 'next/cache'

export async function buildSet(...) {
  const result = await mcpCall('build_set', args)
  revalidateTag('sets') // bust ISR cache
  return result
}

export async function classifyMood(...) {
  const result = await mcpCall('classify_mood', args)
  revalidateTag('library-stats')
  revalidateTag('tracks')
  return result
}

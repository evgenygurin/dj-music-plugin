FastMCP 3.2 — HTTP endpoint (новое!)

# app/http_server.py — FastAPI + FastMCP mount
from fastapi import FastAPI
from app.server import mcp  # existing FastMCP server

mcp_app = mcp.http_app(path="/")
api = FastAPI(title="DJ Music API", lifespan=mcp_app.lifespan)

@api.get("/api/health")
def health():
    return {"status": "ok", "tools": 50}

api.mount("/mcp", mcp_app)

# uvicorn app.http_server:api --host 0.0.0.0 --port 8000
# MCP endpoint: http://localhost:8000/mcp/
# Health: http://localhost:8000/api/health

Next.js → MCP HTTP call (Server Action)

// panel/actions/set-actions.ts
'use server'
import { mcpCall } from '@/lib/mcp-client'

export async function buildSet(playlistId: number, template: string) {
  return mcpCall('build_set', {
    playlist_id: playlistId,
    template,
    algorithm: 'ga',
  })
}

// panel/lib/mcp-client.ts
const MCP_URL = process.env.MCP_HTTP_URL ?? 'http://localhost:8000/mcp'

export async function mcpCall(tool: string, args: Record<string, unknown>) {
  // MCP streamable-http protocol call
  const res = await fetch(`${MCP_URL}/`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      jsonrpc: '2.0', method: 'tools/call',
      params: { name: tool, arguments: args }, id: 1,
    }),
  })
  const data = await res.json()
  return data.result
}

```python
class DJMusicError(Exception):
    """Base error for all DJ Music Plugin errors."""

    def __init__(self, message: str = "", ctx: dict | None = None) -> None:
        self.ctx = ctx or {}
        super().__init__(message)

    def get_context_chain(self) -> dict:
        """Collect ctx from the entire __cause__ chain."""
        result = self.ctx.copy()
        if self.__cause__ and isinstance(self.__cause__, DJMusicError):
            result.update(self.__cause__.get_context_chain())
        return result
```

# src/dj_music/core/logging.py
"""Structured logging configuration with structlog."""

import logging
import sys
from typing import Any

import structlog

def process_custom_exceptions(
    logger: Any, method_name: str, event_dict: dict
) -> dict:
    """Auto-extract ctx from DJMusicError instances."""
    exc_info = event_dict.get("exc_info")
    if exc_info and isinstance(exc_info, tuple):
        exc = exc_info[1]
        from dj_music.core.errors import DJMusicError
        if isinstance(exc, DJMusicError):
            event_dict.update(exc.get_context_chain())
    return event_dict

def setup_logging(*, json: bool = False, level: str = "INFO") -> None:
    """Configure structlog + stdlib logging."""
    shared_processors = [
        structlog.contextvars.merge_contextvars,
        structlog.stdlib.add_log_level,
        structlog.stdlib.add_logger_name,
        process_custom_exceptions,
        structlog.processors.TimeStamper(fmt="iso"),
    ]

    if json:
        renderer = structlog.processors.JSONRenderer()
    else:
        renderer = structlog.dev.ConsoleRenderer()

    structlog.configure(
        processors=[
            *shared_processors,
            structlog.stdlib.ProcessorFormatter.wrap_for_formatter,
        ],
        logger_factory=structlog.stdlib.LoggerFactory(),
    )

    formatter = structlog.stdlib.ProcessorFormatter(
        processors=[*shared_processors, renderer],
    )

    handler = logging.StreamHandler(sys.stdout)
    handler.setFormatter(formatter)

    root = logging.getLogger()
    root.handlers.clear()
    root.addHandler(handler)
    root.setLevel(level)

    # Silence noisy libraries
    for name in ("httpx", "httpcore", "urllib3"):
        logging.getLogger(name).setLevel(logging.WARNING)
```

# src/dj_music/core/observability.py
"""Sentry + OTEL integration."""

import logging
from typing import Callable

logger = logging.getLogger(__name__)

def setup_sentry(dsn: str | None) -> Callable | None:
    """Initialize Sentry SDK. Returns error callback for middleware."""
    if not dsn:
        return None
    try:
        import sentry_sdk
        sentry_sdk.init(dsn=dsn, traces_sample_rate=0.1)
        logger.info("Sentry initialized")
        return lambda exc: sentry_sdk.capture_exception(exc)
    except ImportError:
        logger.warning("sentry-sdk not installed, skipping")
        return None

def setup_otel(endpoint: str | None) -> None:
    """Initialize OpenTelemetry (optional)."""
    if not endpoint:
        return
    try:
        from opentelemetry import trace
        from opentelemetry.sdk.trace import TracerProvider
        trace.set_tracer_provider(TracerProvider())
        logger.info("OTEL initialized", endpoint=endpoint)
    except ImportError:
        logger.warning("opentelemetry not installed, skipping")
```

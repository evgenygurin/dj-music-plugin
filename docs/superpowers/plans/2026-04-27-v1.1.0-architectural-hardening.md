**Goal:** Replace bug-by-bug patching of MCP-client transport asymmetry with three architectural-level safeguards: (1) server-side JSON-string coercion middleware, (2) stringified-args test fixture, (3) integration round-trip tests against real YM (skip-if-no-token).

**Architecture:** All three changes are additive. The middleware sits in the existing `app/server/middleware/` chain and inspects each tool's `inputSchema` to coerce stringified `array`/`object` args before Pydantic validation. The test fixture wraps `fastmcp.Client.call_tool` to JSON-stringify dict/list args, parametrizing existing transport-sensitive tests over both modes. Integration tests live under `tests/providers/yandex/` with `pytest.mark.skipif(not DJ_YM_TOKEN)` so CI without secrets stays green.

**Tech Stack:** FastMCP v3 middleware lifecycle (`on_call_tool`), Pydantic v2 BeforeValidator semantics, pytest-asyncio, pytest fixtures + parametrize, `httpx` against `api.music.yandex.net` (already wired through `YandexClient`).

```python
# app/server/middleware/json_string_coerce.py
"""Server-side coercion of stringified array/object tool args.

Some MCP transports (notably Claude Code's stdio shim) JSON-stringify
``list``/``dict`` arguments before sending. Per-param ``BeforeValidator``
helpers (``app/shared/types.py:JsonIntList`` etc.) already solve this on
a tool-by-tool basis but require explicit opt-in. This middleware is the
opt-out variant: any tool whose ``inputSchema`` declares an arg as
``type: array`` or ``type: object`` will have stringified payloads
parsed back to native types before Pydantic validation runs.
"""

from __future__ import annotations

import json
from typing import Any

from fastmcp.server.middleware import CallNext, Middleware, MiddlewareContext
from mcp.types import CallToolRequestParams, CallToolResult

def _arg_expects_complex(schema: dict[str, Any] | None) -> bool:
    """True if the parameter schema accepts ``array`` or ``object``."""
    if not isinstance(schema, dict):
        return False
    t = schema.get("type")
    if t in {"array", "object"}:
        return True
    for branch_key in ("anyOf", "oneOf", "allOf"):
        branches = schema.get(branch_key)
        if isinstance(branches, list):
            for b in branches:
                if _arg_expects_complex(b):
                    return True
    return False

def _coerce_args_against_schema(
    args: dict[str, Any], input_schema: dict[str, Any] | None
) -> dict[str, Any]:
    """Parse JSON-encoded strings for args whose schema expects array/object."""
    if not isinstance(input_schema, dict):
        return args
    props = input_schema.get("properties")
    if not isinstance(props, dict):
        return args

    out = dict(args)
    for name, prop_schema in props.items():
        if name not in out:
            continue
        if not isinstance(out[name], str):
            continue
        if not _arg_expects_complex(prop_schema):
            continue
        s = out[name].strip()
        if not s or not (s.startswith("[") or s.startswith("{")):
            continue
        try:
            parsed = json.loads(s)
        except json.JSONDecodeError:
            continue
        if isinstance(parsed, (list, dict)):
            out[name] = parsed
    return out

class JsonStringCoerceMiddleware(Middleware):
    """Coerces stringified array/object args at the MCP boundary.

    Runs before Pydantic validation in the tool dispatch chain. Idempotent
    on native types — passes them through unchanged.
    """

    async def on_call_tool(
        self,
        context: MiddlewareContext[CallToolRequestParams],
        call_next: CallNext[CallToolRequestParams, CallToolResult],
    ) -> CallToolResult:
        params = context.message
        tool_name = params.name
        if tool_name and context.fastmcp_context is not None:
            try:
                tool = await context.fastmcp_context.fastmcp.get_tool(tool_name)
            except Exception:
                tool = None
            input_schema = getattr(tool, "parameters", None) if tool else None
            if isinstance(input_schema, dict) and isinstance(params.arguments, dict):
                params.arguments = _coerce_args_against_schema(
                    params.arguments, input_schema
                )
        return await call_next(context)
```

# tests/conftest.py — append at end
import json as _json
from collections.abc import Mapping, Sequence

class _StringifyingClientProxy:
    """Wraps a FastMCP Client; JSON-stringifies dict/list args before call_tool.

    Reproduces the Claude Code stdio shim's transport quirk so that
    in-memory tests catch the same regressions a real client would.
    """

    def __init__(self, inner: Any) -> None:
        self._inner = inner

    def __getattr__(self, name: str) -> Any:
        return getattr(self._inner, name)

    async def __aenter__(self) -> "_StringifyingClientProxy":
        await self._inner.__aenter__()
        return self

    async def __aexit__(self, *exc: Any) -> Any:
        return await self._inner.__aexit__(*exc)

    async def call_tool(self, name: str, args: Mapping[str, Any] | None = None) -> Any:
        coerced: dict[str, Any] = {}
        for k, v in (args or {}).items():
            if isinstance(v, (Mapping, Sequence)) and not isinstance(v, (str, bytes)):
                coerced[k] = _json.dumps(v)
            else:
                coerced[k] = v
        return await self._inner.call_tool(name, coerced)

@pytest.fixture
def stringified_mcp_client(mcp_client: Any) -> _StringifyingClientProxy:
    """Drop-in replacement for ``mcp_client`` that stringifies complex args."""
    return _StringifyingClientProxy(mcp_client)
```

# tests/providers/yandex/test_yandex_integration.py
"""Integration round-trip tests against api.music.yandex.net.

Skipped automatically when ``DJ_YM_TOKEN`` is not set (CI without
secrets, contributor laptops without YM accounts). Covers the surfaces
where v1.0.10-v1.0.12 found output Union-narrowing bugs.
"""

from __future__ import annotations

import os

import pytest
from fastmcp import Client, FastMCP

from app.server.app import build_mcp_server

pytestmark = [
    pytest.mark.integration,
    pytest.mark.skipif(
        not os.environ.get("DJ_YM_TOKEN"),
        reason="DJ_YM_TOKEN not set — integration tests against YM skipped",
    ),
]

@pytest.fixture
async def mcp_with_yandex() -> FastMCP:
    return await build_mcp_server()

@pytest.mark.asyncio
async def test_provider_search_round_trip(mcp_with_yandex: FastMCP) -> None:
    async with Client(mcp_with_yandex) as client:
        result = await client.call_tool(
            "provider_search",
            {"provider": "yandex", "query": "Amelie Lens", "type": "tracks", "limit": 3},
        )
        data = result.structured_content or result.data
        assert data["provider"] == "yandex"
        assert isinstance(data.get("items"), list)

@pytest.mark.asyncio
async def test_provider_read_likes_returns_dict(mcp_with_yandex: FastMCP) -> None:
    async with Client(mcp_with_yandex) as client:
        result = await client.call_tool(
            "provider_read", {"provider": "yandex", "entity": "likes"}
        )
        data = result.structured_content or result.data
        assert isinstance(data["data"], dict)
        assert "track_ids" in data["data"]

@pytest.mark.asyncio
async def test_provider_write_playlist_round_trip(mcp_with_yandex: FastMCP) -> None:
    """Create + delete throwaway playlist. Pins delete returning bare 'ok'.

    Regression for v1.0.12 ProviderWriteResult.data Union (which previously
    crashed with dict_type ValidationError on this exact code path).
    """
    async with Client(mcp_with_yandex) as client:
        created = await client.call_tool(
            "provider_write",
            {
                "provider": "yandex",
                "entity": "playlist",
                "operation": "create",
                "params": {"title": "__pytest_integration_throwaway__"},
            },
        )
        playlist_kind = (created.structured_content or created.data)["data"]["kind"]
        deleted = await client.call_tool(
            "provider_write",
            {
                "provider": "yandex",
                "entity": "playlist",
                "operation": "delete",
                "params": {"playlist_id": playlist_kind},
            },
        )
        result_data = (deleted.structured_content or deleted.data)["data"]
        assert result_data == "ok"
```
